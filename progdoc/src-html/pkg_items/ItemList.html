001package pkg_items;
002import java.util.HashMap;
003import java.util.Set;
004/**
005 * ItemList Class - Holds all the inventory functionalities
006 * @author Aman GHAZANFAR
007 * @version 27.03.2023
008 */
009public class ItemList {
010  private HashMap < String, Item > aList; // hold items
011  /**
012   * Constructor of ItemList Class, creates the HashMap that holds all the items of Player's inventory or Room's item
013   */
014  public ItemList() {
015    this.aList = new HashMap < String, Item > ();
016  } // ItemList()
017  /**
018   * @param pName Item's name
019   * @return Item's corresponding object
020   */
021  public Item getItem(final String pName) {
022    return this.aList.get(pName.toLowerCase());
023  } // getItem(.)
024
025  /**
026   * @return All the items (name and price) of Player's inventory
027   */
028  public String getItemsString() {
029    StringBuilder vParse = new StringBuilder();
030    Set < String > vKeys = this.aList.keySet();
031
032    for (String vItem: vKeys) {
033      vParse.append("[" + vItem + "]" + "(" + this.getItem(vItem).getWeight() + ") | ");
034    }
035    if (vParse.length() == 0) return "No item here";
036    return vParse.toString();
037  } // getItemsString()
038
039  /**
040   * Adds the item to Player's inventory
041   * @param pS The name of the item
042   * @param pI Item's object
043   */
044  public void setItem(final String pS, final Item pI) {
045    this.aList.put(pS.toLowerCase(), pI);
046  } // setItem(.)
047  /**
048   * Removes the item from Player's inventory
049   * @param pName The name of the item
050   */
051  public void removeItem(final String pName) {
052    this.aList.remove(pName);
053  } // removeItem(.)
054
055  @Override public String toString() {
056      StringBuilder vParse = new StringBuilder();
057    Set < String > vKeys = this.aList.keySet();
058
059    for (String vItem: vKeys) {
060        vParse.append(vItem + ",");
061    }
062    return String.join(",",vKeys);
063  }
064} // ItemList
