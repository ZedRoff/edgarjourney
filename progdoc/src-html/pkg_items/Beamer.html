001package pkg_items;
002import pkg_rooms.Room;
003/**
004 * Beamer Class - Handles the methods for a Beamer, Item for teleporting the player
005 * @author Aman GHAZANFAR
006 * @version 2023.04.02
007 */
008public class Beamer extends Item {
009    // Attributes
010    private Room aChargedRoom; // the room the beamer was charge in
011 /**
012     * Constructor, initialize the Beamer like a usual Item
013     * @param pName The name of the beamer
014     * @param pDescription The description of the beamer
015     * @param pWeight The weight of the beamer
016     */
017    public Beamer(final String pName, final String pDescription, final double pWeight) {
018        super(pName, pDescription, pWeight);
019        
020    } // Beamer(.)
021    /**
022     * Changes the Room the Beamer is charged in
023     * @param pRoom The Room where the beamer is charged
024     * @return The String that permits GameEngine to show the result of the request for the command
025     */
026    public String charge(final Room pRoom) {
027            this.aChargedRoom = pRoom;
028            return "You're Beamer has been charged successfuly";
029        
030        
031        } // charge(.)
032        /**
033         * Changes the chargedRoom to null, and teleports the Player into the given room
034         * @return The String that permits GameEngine to show the result of the request for the command
035         */
036    public String fire() {
037        Room vStock = this.aChargedRoom;
038        this.aChargedRoom = null;
039        return "You fired the Beamer, you got teleported to : "+"\n"+vStock.getLongDescription();
040    } // fire()
041    /**
042     * @return The Room the beamer was charged in
043     */
044    public Room getChargedRoom() {
045        return this.aChargedRoom;
046    } // getChargedRoom()
047    /**
048     * @return The state of the Beamer, whether it is charged or not
049     */
050    public boolean canFire() {
051        return this.aChargedRoom != null;
052    } // canFire()
053   
054} // Beamer
