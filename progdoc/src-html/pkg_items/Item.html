001package pkg_items;
002/**
003 * Item Class - Holds all the functionalities of a single item that is inside any Room
004 * @author Aman GHAZANFAR
005 * @version 2023.03.27
006 */
007public class Item {
008  private String aName; // the name of the item
009  private String aDescription; // his description
010  private double aWeight; // and his weight
011  /**
012   * Item Class Constructor, instantiates the attributes of an item
013   * @param pName The name of the item
014   * @param pDescription The description of the item
015   * @param pWeight The weight of the item
016   */
017  public Item(final String pName, final String pDescription, final double pWeight) {
018    this.aName = pName;
019    this.aDescription = pDescription;
020    this.aWeight = pWeight;
021  } // Item(.)
022  /**
023   * @return Item's name
024   */
025  public String getName() {
026    return this.aName;
027  } // getName()
028
029  /**
030   * @return Item's description
031   */
032  public String getDescription() {
033    return this.aDescription;
034  } // getDescription()
035  /**
036   * @return Item's weight
037   */
038  public double getWeight() {
039    return this.aWeight;
040  } // getWeight()
041
042} // Item
