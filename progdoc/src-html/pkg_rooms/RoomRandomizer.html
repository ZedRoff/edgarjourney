001package pkg_rooms;
002
003import java.util.Random;
004
005import java.util.Collection;
006/**
007 * RoomRandomizer Class - Permits to generate a random room, has only one method
008 * @author Aman GHAZANFAR
009 * @version 2023.04.05
010 */
011public class RoomRandomizer {
012    // Attributes
013    private Random aRandom; // Handles a Random object, so we can generate a random number from it
014    /**
015     * Constructor, instantiates the aRandom attribute
016     */
017    public RoomRandomizer() {
018        this.aRandom = new Random();
019        
020    } // RoomRandomizer()
021    /**
022     * @param pRooms The Collection of visitableRooms
023     * @return A random room among the visitable Rooms
024     */
025    public Room generateRandomRoom(final Collection<Room> pRooms) {
026
027        int vRnd = this.aRandom.nextInt(pRooms.size());
028       return (Room)pRooms.toArray()[vRnd];
029       
030    } // generateRandomRoom(.)
031  
032} // RoomRandomizer
