001package pkg_gameobjects;
002import pkg_items.Item;
003
004/**
005 * Door Class - Creates doors, so we can lock them
006 * @author Aman GHAZANFAR
007 * @version 2023.04.02
008 */
009 public class Door {
010    // Attributes
011    private Item aKey; // The key that is necessary to open the Door
012    private boolean aState; // says whether the door is opened or not
013    /**
014     * Constructor, instantiates the state of the door (whether it's locked or not).
015     */
016    public Door() {
017        this.aState = false;
018    } // Door()
019    /**
020     * Handles the lock of a Door
021     * @param pKey The Key item in order to open the door
022     */
023    public void lock(final Item pKey) {
024        this.aState = true;
025        this.aKey = pKey;
026    } // lock(.)
027    /**
028     * @return A boolean that contains the state of the Door (opened or closed)
029     */
030    public boolean isLocked() {
031        return this.aState;
032    } // isLocked()
033    /**
034     * @return The Key that is required to open the Door
035     */
036    public Item getKey() {
037        return this.aKey;
038    } // getKey()
039} // Door
