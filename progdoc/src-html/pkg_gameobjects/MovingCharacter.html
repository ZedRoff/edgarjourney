001package pkg_gameobjects;
002import pkg_rooms.Room;
003/**
004 * MovingCharacter Class - Handles all the moving NPCs of the Game
005 */
006public class MovingCharacter extends Character {
007    // Attributes
008    private Room aCurrentRoom;
009   
010    /**
011     * Character Class Constructor, instantiates all the attributes
012     * @param pName NPC's name
013     * @param pDescriptipn NPC's description
014     * @param pDialog NPC's dialog
015     */
016    public MovingCharacter(final String pName, final String pDescription, final String pDialog) {
017        super(pName, pDescription, pDialog);
018    } // MovingCharacter(.)
019    /**
020     * Allowed a MovingCharacter type object to move around the map
021     * @param pRoom The new Room he is going to be in
022     */
023    public void move(final Room pRoom) {
024        aCurrentRoom.removeCharacter(this);
025        pRoom.addCharacter(this);
026        aCurrentRoom = pRoom;
027    } // move(.)
028    
029} // MovingCharacter(.)
