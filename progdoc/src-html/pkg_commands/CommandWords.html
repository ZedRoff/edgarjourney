001package pkg_commands;
002import java.util.HashMap;
003 import java.util.Set;
004/**
005
006  * This class holds an enumeration table of all command words known to the game.
007  * It is used to recognise commands as they are typed in.
008  *
009  * @author Aman GHAZANFAR
010  * @version 2023.02.09
011  */
012 public class CommandWords {
013   // Attributes
014   
015   private HashMap<String, CommandWord> aValidCommands; // a constant array that will hold all valid command words
016  /**
017   * Constructor, instantiates the aValidCommands attribute by searching inside the CommandWord Class' enums
018   */
019   public CommandWords() {
020       aValidCommands = new HashMap<String, CommandWord>();
021      
022        for(CommandWord command : CommandWord.values()) {
023            if(command != CommandWord.UNKNOWN) {
024                aValidCommands.put(command.toString(), command);
025            }
026        }
027   } // CommandWords()
028    
029    /**
030    * @param pCommandWord The first word of the command typed in
031    * @return The command corresponding to the pCommandWord given
032    */
033    public CommandWord getCommandWord(String pCommandWord)
034    {
035        CommandWord command = aValidCommands.get(pCommandWord);
036        if(command != null) {
037            return command;
038        }
039        else {
040            return CommandWord.UNKNOWN;
041        }
042    } // CommandWord(.)
043    
044    
045   /**
046    * @return The list of all commands inside a single String
047    */
048   public String getCommandList() {
049     StringBuilder vCommands = new StringBuilder();
050     Set<String> vKeys = aValidCommands.keySet();
051     for (String command: vKeys) {
052       vCommands.append(command + " ");
053
054     }
055     return vCommands.toString();
056   } // getCommandList()
057
058   /**
059    * Check whether a given String is a valid command word. 
060    * @param pString The command slitted with a space
061    * @return true if a given string is a valid command, false if it isn't.
062    */
063   public boolean isCommand(final String pString) {
064     return aValidCommands.containsKey(pString);
065   } // isCommand(.)
066   /**
067    * @param pCommandWord The first word of any command
068    * @return returns the actual function corresponding to the commandWord given
069    */
070   public Command getFunc(final String pCommandWord) {
071       return this.getCommandWord(pCommandWord).toClass();
072   } // getFunc(.)
073 } // CommandWords
