001package pkg_commands;
002import pkg_gameobjects.Player;
003import pkg_items.Item;
004/**
005 * DropCommand Class - Permits to drop an item inside a Room
006 * @author Aman GHAZANFAR 
007 * @version 2023/05/06
008 */
009public class DropCommand extends Command {
010     /**
011     * Execute function, common to every commands, overrides the abstract method of Command Class.
012     * @param pPlayer A Player object
013     */
014    @Override
015    public void execute(Player pPlayer) {
016        
017        // check if there is a second word
018         if (!super.hasSecondWord()) pPlayer.getGUI().println("This command requires a second argument (item)");
019      else {
020          // check if the user has the item he wants to drop
021   
022       
023       if(super.getSecondWord().equals("coin")) {
024           if(pPlayer.getCoins() > 0) {
025                    pPlayer.removeCoins();
026         pPlayer.getGUI().println("You dropped a coin, you now have : "+pPlayer.getCoins()); // success message
027          pPlayer.getCurrentRoom().addItem(new Item("coin", "A coins, let's see its purpose", 1)); // add the item into the current room
028           }else {
029               pPlayer.getGUI().println("You can't drop a coin, since you don't have one");
030           }
031   
032         return;
033    }
034    if (!pPlayer.hasItem(super.getSecondWord())) {
035       pPlayer.getGUI().println("You don't have this item");
036       return;
037    };
038    
039    Item vItem = pPlayer.getInventory().getItem(super.getSecondWord()); // get the item he wants to drop
040   
041      
042    
043    
044    pPlayer.getCurrentRoom().addItem(vItem); // add the item into the current room
045    pPlayer.setCurrentWeight(pPlayer.getCurrentWeight() - vItem.getWeight()); // change his weight
046    
047   
048    pPlayer.getInventory().removeItem(vItem.getName()); // remove the item from his inventory
049    pPlayer.getGUI().println("You drop the " + super.getSecondWord() + ". Your new total weight is : " + pPlayer.getCurrentWeight() + "kg"); // show him the success message
050      }
051 
052   
053    } // execute(.)
054} // DropCommand
