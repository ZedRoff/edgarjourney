001 package pkg_commands;
002 import pkg_gameobjects.Player;
003/**
004  * Command Class - Handles all the command side, a command is composed of two words (the command word and the argument word).
005  *
006  * @author Aman GHAZANFAR
007  * @version 2023.02.09
008  */
009 public abstract class Command {
010   // Attributes
011   private String aSecondWord; // the second word of this command typed in
012 private String aCommandWord;
013
014
015   /**
016    * @return A String with the argument word. (second word).
017    */
018   public String getSecondWord() {
019     return this.aSecondWord;
020   } // getSecondWord()
021    public boolean isUnknown(){
022        return this.getCommandWord() == null;
023    }
024   /**
025    * @return The command word of the line user gave
026    */
027       public String getCommandWord(){
028        return this.aCommandWord;
029    } // getCommandWord()
030   /**
031    * @return A boolean : true if there is a second word. false : if there is no second word.
032    */
033   public boolean hasSecondWord() {
034     return this.aSecondWord != null;
035   } // hasSecondWord()
036    /**
037     * Define the second word of this command (the word
038     * entered after the command word). Null indicates that 
039     * there was no second word.
040     * @param pSecondWord The second word of any Command
041     */
042    public void setSecondWord(String pSecondWord)
043    {
044        this.aSecondWord = pSecondWord;
045    } // setSecondWord(.)
046    /**
047     * Sets the command word on the command (the word entered before the second word). 
048     * Null indicates that there was no first word.
049     * @param pCommandWord The first word of any command
050     */
051    public void setCommandWord(String pCommandWord) {
052        this.aCommandWord = pCommandWord;
053    } // setCommandWord(.)
054  /**
055   * Abstract method, has to be re-written inside every command, permits to avoid using all the switch inside interpretCommand, heritage 
056   * @param pPlayer The Player Object (current Player so it can handle more than 1 player)
057   */
058    public abstract void execute(final Player pPlayer);
059    //execute(.)
060 
061 } // Command
