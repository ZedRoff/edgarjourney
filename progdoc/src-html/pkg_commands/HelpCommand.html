001package pkg_commands;
002import pkg_gameobjects.Player;
003/**
004 * HelpCommand Class - Shows the help manual to the player
005 */
006public class HelpCommand extends Command {
007   /**
008     * Execute function, common to every commands, overrides the abstract method of Command Class.
009     * @param pPlayer A Player object
010     */
011    @Override
012    public void execute(Player pPlayer) {
013         pPlayer.getGUI().println("How could you be lost inside/near your own house..., anyway, here is the help page\n.You command word are: ");
014    pPlayer.getGUI().println(pPlayer.getGameEngine().getParser().getCommandString()); // the help page is here
015    } // execute(.)
016} // HelpCommand
