001package pkg_commands;
002import pkg_gameobjects.Player;
003import pkg_rooms.TransporterRoom;
004/**
005 * AleaCommand Class - Command only usable while in test mode, be sure to change the option in GameEngine before trying it.
006 * Permits to alter the TransporterRoom exit (making it non-random, for testing purpose).
007 * @author Aman GHAZANFAR
008 * @version 2023/05/06
009 */
010public class AleaCommand extends Command {
011     /**
012     * Execute function, common to every commands, overrides the abstract method of Command Class.
013     * @param pPlayer A Player object
014     */
015    @Override
016    public void execute(Player pPlayer) {
017       // gets the transporter Room
018        TransporterRoom vTransporter = (TransporterRoom)pPlayer.getGameEngine().getRooms().get("transporter");
019        
020        if(pPlayer.getTestMode()) { // check if the player is in test mode
021            if(!super.hasSecondWord()) { // check if there is a second word
022                vTransporter.setAlea(null); // sets the transporterRoom to default value
023              pPlayer.getGUI().println("The default mode is actived, now you are going to be teleported to a random room once exiting the transporter room"); // success message
024            }else { // if there is a second word
025                if(pPlayer.getGameEngine().getRooms().get(super.getSecondWord()) == null) { // if the input Room doesn't exists
026                    pPlayer.getGUI().println("This room wasn't recognized by the system");
027                    return;
028                } // if it exists
029                vTransporter.setAlea(super.getSecondWord()); // changes the random room to the given one
030                pPlayer.getGUI().println("Ok, now you will be teleported to "+super.getSecondWord()+" once exiting the transporter room"); // success message
031            }
032        } else { // if the player isn't in test mode
033            pPlayer.getGUI().println("You aren't in test mode");
034        }
035    } // execute(.)
036} // AleaCommand
