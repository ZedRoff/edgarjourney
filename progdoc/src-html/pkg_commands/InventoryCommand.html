001package pkg_commands;
002import pkg_gameobjects.Player;
003
004/**
005 * InventoryCommand Class - Shows user's inventory
006 * @author Aman GHAZANFAR
007 * @version 2023/05/06
008 */
009public class InventoryCommand extends Command {
010  
011     /**
012     * Execute function, common to every commands, overrides the abstract method of Command Class.
013     * @param pPlayer A Player object
014     */
015    @Override
016    public void execute(Player pPlayer) {
017        
018          pPlayer.getGUI().println(pPlayer.getInventoryString()); // pretty explicit
019          pPlayer.getGUI().println("You can carry on more : "+(pPlayer.getMaxWeight()-pPlayer.getCurrentWeight())+"kg worth of items");
020          pPlayer.getGUI().println("You also have : "+pPlayer.getCoins()+" coins");
021          
022    } // execute(.)
023} // InventoryCommand
