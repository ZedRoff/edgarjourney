001package pkg_commands;

003
004 import javax.swing.JFrame;
005import javax.swing.ImageIcon;
006import javax.swing.JLabel;
007
008import pkg_gameobjects.Player;
009import pkg_gameobjects.Character;
010import pkg_gameobjects.MovingCharacter;
011/**
012 * TalkCommand Class - Permits the player to talk to an NPC
013 * @author Aman GHAZANFAR
014 * @version 2023/05/06
015 */
016public class TalkCommand extends Command {
017     /**
018     * Execute function, common to every commands, overrides the abstract method of Command Class.
019     * @param pPlayer A Player object
020     */
021    @Override
022    public void execute(Player pPlayer) {
023        
024        // check if there is a second word
025        if (!super.hasSecondWord()) {
026      pPlayer.getGUI().println("Talk to who ?");
027      return;
028    }
029    Character vCharacter = pPlayer.getCurrentRoom().getCharacter(super.getSecondWord()); // get the NPC by its name
030    if(vCharacter instanceof MovingCharacter) {
031        JFrame vFrame = new JFrame("The Map");
032    ImageIcon vIcon = new ImageIcon("./assets/map.png");
033    JLabel vLabel = new JLabel(vIcon);
034    vFrame.add(vLabel);
035    vFrame.pack();
036    vFrame.setVisible(true);
037    }
038    if(vCharacter == null) { // if the NPC isn't inside the Room
039        pPlayer.getGUI().println("There is no such NPC inside this Room");
040        return;
041    }
042    pPlayer.getGUI().println(vCharacter.getName() + " said : "+vCharacter.getDialog());
043    
044   
045    
046    } // execute(.)
047} // GoCommand
