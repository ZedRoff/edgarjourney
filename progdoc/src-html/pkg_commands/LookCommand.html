001 package pkg_commands;
002import pkg_gameobjects.Player;
003import pkg_items.Item;
004/**
005 * LookCommand Class - Permits to either look inside the room, or look into a specific item (whether you give it an argument or not)
006 * @author Aman GHAZANFAR
007 * @version 2023/05/06
008 */
009public class LookCommand extends Command {
010     /**
011     * Execute function, common to every commands, overrides the abstract method of Command Class.
012     * @param pPlayer A Player object
013     */
014    @Override
015    public void execute(Player pPlayer) {
016     // check if there is a second word
017       if (super.hasSecondWord()) {
018      Item vItem = pPlayer.getCurrentRoom().getItem(super.getSecondWord());
019      
020      // check if the item he wants to look into is inside the room or not
021      if (vItem == null) {
022        pPlayer.getGUI().println("This item doesn't exists");
023        return;
024      }
025      
026      // show item's caracteristics
027     pPlayer.getGUI().println("Item's name : " + vItem.getName());
028      pPlayer.getGUI().println("Item's description : " + vItem.getDescription());
029      pPlayer.getGUI().println("Item's weight : " + vItem.getWeight());
030    } else {
031      // show room's caracteristics
032     pPlayer.getGUI().println(pPlayer.getCurrentRoom().getLongDescription());
033    }
034    } // execute(.)
035} // LookCommand
