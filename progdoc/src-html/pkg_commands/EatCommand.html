001package pkg_commands;
002import pkg_gameobjects.Player;
003/**
004 * EatCommand Class - Permits player the eat for instance a cookie
005 * @author Aman GHAZANFAR
006 * @version 2023/05/06
007 */
008public class EatCommand extends Command {
009    /**
010     * Execute function, common to every commands, overrides the abstract method of Command Class.
011     * @param pPlayer A Player object
012     */
013    @Override
014    public void execute(Player pPlayer) {
015       String[] vIngredients = {
016      "cookie"
017    };
018
019    boolean verif = false;
020    
021    // checks if there is a second word
022  
023    if (!super.hasSecondWord()) {
024       pPlayer.getGUI().println( "Eat what ?");
025    } else {
026
027        // checks if the choice of user is an actual ingrediant that is eatable
028      StringBuilder parse = new StringBuilder();
029      for (String ingredient: vIngredients) {
030        if (ingredient.equals(super.getSecondWord())) {
031          verif = true;
032
033        }
034        parse.append(" " + ingredient);
035
036      }
037      // if eatable
038      if (verif) {
039        if (pPlayer.hasItem("cookie")) { // if it's the magic cookie, and he has it
040          pPlayer.setMaxWeight(pPlayer.getMaxWeight() + 10); // permit the player to carry on more weight
041         pPlayer.getGUI().println("You ate a " + super.getSecondWord() + " you can now lift items up to " + pPlayer.getMaxWeight());
042
043        } else { // if it's the magic cookie, but he doesn't have it
044           pPlayer.getGUI().println("You don't have this item, but it exists");
045        }
046
047      } else { // if not
048       pPlayer.getGUI().println("Wrong ingredient, you can only eat : " + parse.toString());
049      }
050
051    }
052    } // execute(.)
053} // EatCommand
