001package pkg_commands;
002import pkg_gameobjects.Player;
003import pkg_items.Beamer;
004import pkg_items.Item;
005/**
006 * FireCommand Class - Permits to fire the Beamer so it teleports you to a Room
007 * @author Aman GHAZANFAR
008 * @version 2023/05/06
009 */
010public class FireCommand extends Command {
011 /**
012     * Execute function, common to every commands, overrides the abstract method of Command Class.
013     * @param pPlayer A Player object
014     */
015    @Override
016    public void execute(Player pPlayer) {
017        
018        // check if there is a second word
019         if(!super.hasSecondWord()) {
020            pPlayer.getGUI().println("Fire what ?");
021        }else {
022    
023        Item vItem = pPlayer.getInventory().getItem(super.getSecondWord()); // get the beamer from player's inventory
024      if(pPlayer.hasItem(super.getSecondWord())) { // check if he has it, so vItem can't be null
025          Beamer vBeamer = (Beamer)vItem; // casts the Item object into a Beamer object so we can use it
026          if(!vBeamer.canFire()) { // if we can't fire the beamer
027             pPlayer.getGUI().println("The Beamer can't fire, please charge it first."); // error message
028          } else { // if we can fire it
029              pPlayer.setCurrentRoom(vBeamer.getChargedRoom()); // teleport the player to the room where the Beamer was initially charged in (include printLocationInfo())
030            pPlayer.getGUI().println(vBeamer.fire()); // shows the teleportation success message
031          }
032      } else {
033           pPlayer.getGUI().println("You don't have the Beamer, find it first."); // error message
034      }  
035if( pPlayer.getCurrentRoom().getImageName() != null )
036            pPlayer.getGUI().showImage( pPlayer.getCurrentRoom().getImageName() );      // shows room's image 
037        }
038    } // execute(.)
039} // FireCommand
