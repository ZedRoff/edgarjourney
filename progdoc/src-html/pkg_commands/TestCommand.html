001package pkg_commands;
002import java.io.File;
003import java.util.Scanner;
004import pkg_gameobjects.Player;
005/**
006 * TestCommand Class - Permits to test the game thanks to a txt file that has a specified path to follow upon commands
007 * @author Aman GHAZANFAR
008 * @version 2023/05/06
009 */
010public class TestCommand extends Command {
011 /**
012     * Execute function, common to every commands, overrides the abstract method of Command Class.
013     * @param pPlayer A Player object
014     */
015    @Override
016    public void execute(Player pPlayer) {
017        
018        // check if there is a second word
019        if(!super.hasSecondWord()) {
020          pPlayer.getGUI().println("Test what ?");
021          return;
022        }
023      File vFile = new File("./tests/" + super.getSecondWord() + ".txt"); // get the file which user wants to try the game with
024    Scanner vScan; // Scanner object thanks to it, we can read files line by line (or sentences word by word)
025    try {
026      vScan = new Scanner(vFile);
027      while (vScan.hasNextLine()) { // while there is a new line, continue reading it
028            String vLine = vScan.nextLine(); // get the next line
029            if(vLine.length() == 0) return; // it there is no carac on that line, no need to run it
030           pPlayer.getGameEngine().interpretCommand(vLine); // run the command on that line
031      }
032      vScan.close(); // close the scanner
033    } catch (Exception FileNotFoundException) { // if the file wasn't found
034      pPlayer.getGUI().println("File doesn't exists");
035    }
036    } // execute(.)
037} // TestCommand
