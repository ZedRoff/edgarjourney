001package pkg_commands;
002
003import pkg_gameobjects.Player;
004import pkg_rooms.Room;
005import pkg_items.Item;
006/**
007 * TalkCommand Class - Permits the player to talk to an NPC
008 * @author Aman GHAZANFAR
009 * @version 2023/05/06
010 */
011public class InteractCommand extends Command {
012     /**
013     * Execute function, common to every commands, overrides the abstract method of Command Class.
014     * @param pPlayer A Player object
015     */
016    @Override
017    public void execute(Player pPlayer) {
018        
019        // check if there is a second word
020        if (!super.hasSecondWord()) {
021      pPlayer.getGUI().println("Talk to who ?");
022      return;
023    }
024    if(pPlayer.getCurrentRoom().getName().equals("cabin")) {
025        if(super.getSecondWord().equals("123")) {
026            pPlayer.getGUI().println("Wrong number, maybe the cat was wrong ? Find another way to get the code...");
027            return;
028        }else if(super.getSecondWord().equals("321")) {
029            pPlayer.grantAccess();
030            pPlayer.getGUI().println("Right code, permission to enter basement granted");
031            return;
032        }
033    } else if(pPlayer.getCurrentRoom().getName().equals("living")) {
034        if(!super.getSecondWord().equals("watch")) {
035            pPlayer.getGUI().println("Wrong answer, try again");
036        }else {
037            pPlayer.getGUI().println("Right answer, you now have access granted to go inside the laboratory");
038            pPlayer.getInventory().setItem("code", new Item("code", "The code of the laboratory now Edgar can use it if you go to this direction", 0));
039        }
040    } else if(pPlayer.getCurrentRoom().getName().equals("laboratory")) {
041        if(!super.getSecondWord().equals("helloworld")) {
042            pPlayer.getGUI().println("Wrong password");
043        } else {
044            if(!pPlayer.hasItem("pass")){
045                javax.swing.JOptionPane.showMessageDialog(null, "Well played ! Edgar can now code, the game has ended. However, there is an item you missed in this game, try to find it :D");
046     return;
047            }
048            javax.swing.JOptionPane.showMessageDialog(null, "Well played ! Edgar can now code, the game has ended. Also, you find the pass, which completes the game.");
049   
050            pPlayer.getGUI().getFrame().dispose(); // disable the frame
051        pPlayer.getGUI().enable(false); // disable the entry field in case 
052        }
053    
054    }else {
055        pPlayer.getGUI().println("This command can only be used inside the cabin & the living room");
056    }
057    } // execute(.)
058} // InteractCommand
