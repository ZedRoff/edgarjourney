001package pkg_commands;

010/**
011 * SaveCommand Class - Permits the player to save the game
012 * @author Aman GHAZANFAR
013 * @version 2023/05/06
014 */
015public class SaveCommand extends Command {
016     /**
017     * Execute function, common to every commands, overrides the abstract method of Command Class.
018     * @param pPlayer A Player object
019     */
020    @Override
021    public void execute(Player pPlayer) {
022        
023        // check if there is a second word
024        if (!super.hasSecondWord()) {
025      pPlayer.getGUI().println("Save where ?");
026      return;
027    }
028    try {
029       FileWriter vFile = new FileWriter("./saves/"+super.getSecondWord()+".txt"); 
030       vFile.write(pPlayer.getActions().toString().replace("[", "").replace("]", ""));
031       vFile.close();
032       pPlayer.getGUI().println("You're game was save successfully");
033    } catch(IOException e) {
034        pPlayer.getGUI().println("The file wasn't found");
035         e.printStackTrace();
036    }
037    
038    
039    
040    } // execute(.)
041} // SaveCommand
