001package pkg_commands;
002
003import pkg_gameobjects.Player;
004/**
005 * QuitCommand Class - Permits to quit the game, shows a dialog right before, action depending on the choice
006 * @author Aman GHAZANFAR
007 * @version 2023/05/06
008 */
009public class QuitCommand extends Command {
010   
011    /**
012     * Execute function, common to every commands, overrides the abstract method of Command Class.
013     * @param pPlayer A Player object
014     */
015    @Override
016    public void execute(Player pPlayer) {
017        
018        // checks whether a second word was input
019          if (super.hasSecondWord()) {
020      pPlayer.getGUI().println("Quit what ?");
021
022    } else {
023     
024        // takes user's name
025    int input = javax.swing.JOptionPane.showConfirmDialog(null, "Are you sure you want to quit ? :/");
026  
027    if (input == 1) { // if he doesn't want to quit
028      return;
029    } else if(input == 0) { // if he wants to quit
030      javax.swing.JOptionPane.showMessageDialog(null, "Thank you for playing. Good bye.");
031      pPlayer.getGUI().getFrame().dispose(); // disable the frame
032        pPlayer.getGUI().enable(false); // disable the entry field in case 
033    }
034    }
035} // execute(.)
036} // QuitCommand
