001package pkg_commands;
002import pkg_gameobjects.Player;
003import pkg_items.Beamer;
004import pkg_items.Item;
005/**
006 * ChargeCommand Class - Permits to charge the Beamer inside a Room
007 * @author Aman GHAZANFAR
008 * @version 2023/05/06
009 */
010public class ChargeCommand extends Command {
011    /**
012     * Execute function, common to every commands, overrides the abstract method of Command Class.
013     * @param pPlayer A Player object
014     */
015    @Override
016    public void execute(Player pPlayer) {
017        // checks if there is a second word
018       if(!super.hasSecondWord()) {
019            pPlayer.getGUI().println("Charge what ?");
020        }else {
021       Item vItem = pPlayer.getInventory().getItem(super.getSecondWord()); // get the item from his inventory
022      if(pPlayer.hasItem(super.getSecondWord())) { // if the player has the beamer
023          Beamer vBeamer = (Beamer)vItem;
024          if(vBeamer.getChargedRoom() != null) { // if the beamer is charged inside a room
025              pPlayer.getGUI().println("You're Beamer is already charged.");
026          } else { // if it isn't charged inside a room
027              pPlayer.getGUI().println(vBeamer.charge(pPlayer.getCurrentRoom())); // shows the success message of the Beamer that is fully charged
028          }
029      } else { // if he doens't have the Beamer
030          pPlayer.getGUI().println("You don't have the Beamer, find it first.");
031      }
032    }
033} // execute(.)
034} // ChargeCommand
